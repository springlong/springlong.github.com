<script>
    var vm = avalon.define({
        $id: "databind",
        name: "龙泉",
        fruit: "火龙果"
    });
</script>
<div ms-controller="databind">
    <p><input ms-duplex="@name"><strong ms-text='@name'></strong></p>
    <p>
        <select ms-duplex='@fruit'>
            <option>苹果</option>
            <option>香蕉</option>
            <option>火龙果</option>
        </select>
        <span>{{@fruit}}</span>
    </p>
</div>
