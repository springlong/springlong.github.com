响应用户行为

<script>
    avalon.define({
        $id: 'behavior',
        fn: function(e) {
            this.text = e.pageX + ' : ' + e.pageY;
        },
        text: ''
    });
</script>
<div ms-controller="behavior">
    <div ms-on-mousemove="@fn" style="width:200px; height: 200px; background: #4298ba;">
        {{@text}}
    </div>
</div>
{{@text}}
