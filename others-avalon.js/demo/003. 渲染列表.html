渲染列表

<script>
    var vm = avalon.define({
        $id: 'listoutput',
        obj: {
            a: 1,
            b: 2,
            c: 3
        },
        arr: ['xx', 'yy', 'zz']
    });
</script>
<div ms-controller="listoutput">
    <ul>
        <li ms-for="(key, val) in @obj">{{key}}::{{val}}</li>
    </ul>
    <ul>
        <li ms-for="(key, val) in @arr">{{key}}::{{val}}</li>
    </ul>
</div>
